-- Авто-фарм скрипт
function autoFarm()
    while true do
        wait(1)
        for _, coin in pairs(workspace.Coins:GetChildren()) do
            if coin:IsA("Part") then
                firetouchinterest(coin, player.Character.HumanoidRootPart, 0)
                wait(0.1)
                firetouchinterest(coin, player.Character.HumanoidRootPart, 1)
            end
        end
    end
end

autoFarm()
